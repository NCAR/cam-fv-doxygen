243!-----------------------------------------------------------------------
244!> Called to do one time initialization of the model.
245!> In this case, it reads in the grid information, the namelist
246!> containing the variables of interest, where to get them, their size,
247!> their associated DART Quantity, etc.
248!>
249!> In addition to harvesting the model metadata (grid,
250!> desired model advance step, etc.), it also fills a structure
251!> containing information about what variables are where in the DART
252!> framework.
253
254subroutine static_init_model()
255
256integer :: iunit, io
257
258character(len=*), parameter :: routine = 'static_init_model'
259
260if ( module_initialized ) return
261
262! Record version info
263call register_module(source, revision, revdate)
264
265module_initialized = .true.
266common_initialized = .true.
267
268! Read the DART namelist for this model
269call find_namelist_in_file('input.nml', 'model_nml', iunit)
270read(iunit, nml = model_nml, iostat = io)
271call check_namelist_read(iunit, io, 'model_nml')
272
273! Record the namelist values used for the run
274if (do_nml_file()) write(nmlfileunit, nml=model_nml)
275if (do_nml_term()) write( * , nml=model_nml)
276
277! Set values from namelist in cam_common_code_mod
278cuse_log_vertical_scale = use_log_vertical_scale
279cno_normalization_of_scale_heights = no_normalization_of_scale_heights
280cdebug_level = debug_level
281ccustom_routine_to_generate_ensemble = custom_routine_to_generate_ensemble
282ccustom_routine_to_generate_ensemble = custom_routine_to_generate_ensemble
283cperturbation_amplitude = perturbation_amplitude
284cassimilation_period_days = assimilation_period_days
285cassimilation_period_seconds = assimilation_period_seconds
286csuppress_grid_info_in_output = suppress_grid_info_in_output
287
288
289call set_calendar_type('GREGORIAN')
290
291call read_grid_info(cam_template_filename)
292! This non-state variable is used to compute surface elevation.
293call read_cam_phis_array(cam_phis_filename)
294call setup_interpolation() !grid is global
295
296! initialize global values that are used frequently
297call init_globals()
298
299! read the namelist &model_nml :: state_variables
300! to set up what will be read into the cam state vector
301call set_cam_variable_info(cam_template_filename, state_variables)
302
303call fill_cam_stagger_info(grid_stagger)
304
305if (debug_level > 100) call state_structure_info(domain_id)
306
307
308! convert from string in namelist to integer (e.g. VERTISxxx)
309! and tell the dart code which vertical type we want to localize in.
310call set_vert_localization(vertical_localization_coord)
311
312! if you have chemistry variables in the model state, set
313! this namelist variable so we can initialize the proper tables
314if (using_chemistry) call init_chem_tables()
315
316! set top limit where obs impacts are diminished to 0.
317! only allowed if doing vertical localization. error if
318! computing horizontal distances only (odd case, intentionally
319! choosing not to support this.)
320if (model_damping_ends_at_level > 0) then
321 if (vertical_localization_on()) then
322 call init_damping_ramp_info(model_damping_ends_at_level)
323 are_damping = .true.
324 else
325 string1='cannot support model top damping unless also using vertical localization'
326 string2='set "model_damping_ends_at_level = -1" in &model_nml, OR'
327 string3='set "horiz_dist_only = .false." in &location_nml'
328 call error_handler(e_err, routine, string1, source, revision, revdate, &
329 text2=string2, text3=string3)
330 endif
331endif
332
333! set top limit where obs are discarded. -1 to disable.
334if (no_obs_assim_above_level > 0) then
335 call init_discard_high_obs(no_obs_assim_above_level)
336 discarding_high_obs = .true.
337endif
338
339! set a flag based on the vertical localization coordinate selected
340call init_sign_of_vert_units()
341
342end subroutine static_init_model

398!-----------------------------------------------------------------------
399!> given the (i,j,k) indices into a field in the state vector,
400!> and the quantity, and the dimensionality of the field (2d, 3d),
401!> compute the location of that item.
402
403function get_location_from_index(i, j, k, q, nd)
404integer, intent(in) :: i
405integer, intent(in) :: j
406integer, intent(in) :: k
407integer, intent(in) :: q
408integer, intent(in) :: nd
409type(location_type) :: get_location_from_index
410
411character(len=*), parameter :: routine = 'get_location_from_index'
412real(r8) :: slon_val
413real(r8) :: use_vert_val
414integer :: use_vert_type
415
416! full 3d fields are returned with lon/lat/level.
417! 2d fields are either surface fields, or if they
418! are column integrated values then they are 'undefined'
419! in the vertical.
420
421if (nd == 3) then
422 use_vert_type = vertislevel
423 use_vert_val = real(k,r8)
424else if (nd == 2) then
425 ! add any 2d surface fields to this function
426 if (is_surface_field(q)) then
427 use_vert_type = vertissurface
428 use_vert_val = missing_r8
429 ! setting the vertical value to missing matches what the previous
430 ! version of this code did. other models choose to set the vertical
431 ! value to the model surface elevation at this location:
432 ! use_vert_val = phis(lon_index, lat_index) / gravity
433 else
434 ! any 2d field not listed as a surface field (in is_surface_field() function)
435 ! is assumed to be an integrated quantity with a vert type of VERTISUNDEF.
436 use_vert_type = vertisundef
437 use_vert_val = missing_r8
438 endif
439else
440 write(string1, *) 'state vector field not 2D or 3D and no code to handle other dimensionity'
441 write(string2, *) 'dimensionality = ', nd, ' quantity type = ', trim(get_name_for_quantity(q))
442 call error_handler(e_err,routine,string1,source,revision,revdate,text2=string2)
443endif
444
445! the horizontal location depends on whether this quantity is on the
446! mass point grid or staggered in either lat or lon.
447
448select case (grid_stagger%qty_stagger(q))
449 case (stagger_u)
450 get_location_from_index = set_location(grid_data%lon%vals(i), &
451 grid_data%slat%vals(j), &
452 use_vert_val, use_vert_type)
453
454 case (stagger_v)
455 ! the first staggered longitude is negative. dart requires lons
456 ! be between 0 and 360.
457 slon_val = grid_data%slon%vals(i)
458 if (slon_val < 0) slon_val = slon_val + 360.0_r8
459 get_location_from_index = set_location(slon_val, &
460 grid_data%lat%vals(j), &
461 use_vert_val, use_vert_type)
462
463 !>@todo not sure what to do yet. ? +-1/2 ?
464 case (stagger_w)
465 get_location_from_index = set_location(grid_data%lon%vals(i), &
466 grid_data%lat%vals(j), &
467 use_vert_val - 0.5_r8, use_vert_type)
468 ! no stagger - cell centers
469 case default
470 get_location_from_index = set_location(grid_data%lon%vals(i), &
471 grid_data%lat%vals(j), &
472 use_vert_val, use_vert_type)
473
474end select
475
476end function get_location_from_index

478!-----------------------------------------------------------------------

480!> unstaggered grid but needs to correspond to a staggered grid.
481!> e.g. you need the surface pressure under a V wind point.

483subroutine get_staggered_values_from_qty(ens_handle, ens_size, qty, lon_index, lat_index, &
484 lev_index, stagger_qty, vals, my_status)
485type(ensemble_type), intent(in) :: ens_handle
486integer, intent(in) :: ens_size
487integer, intent(in) :: qty
488integer, intent(in) :: lon_index
489integer, intent(in) :: lat_index
490integer, intent(in) :: lev_index

493integer, intent(out) :: my_status
494
495integer :: next_lat, prev_lon, stagger

498vals(:) = missing_r8
499stagger = grid_stagger%qty_stagger(stagger_qty)
500
501!> latitudes: staggered value N is between N and (N + 1) on the unstaggered grid
502!> longitudes: staggered value N is between N and (N - 1) on the unstaggered grid
503
504select case (stagger)
505 case (stagger_u)
506 call quad_index_neighbors(lon_index, lat_index, prev_lon, next_lat)
507
508 call get_values_from_single_level(ens_handle, ens_size, qty, lon_index, lat_index, lev_index, &

510 if (my_status /= 0) return
511 call get_values_from_single_level(ens_handle, ens_size, qty, lon_index, next_lat, lev_index, &
512 vals2, my_status)
513 if (my_status /= 0) return
514
515 vals = (vals1 + vals2) * 0.5_r8
516
517 case (stagger_v)
518 call quad_index_neighbors(lon_index, lat_index, prev_lon, next_lat)
519
520 call get_values_from_single_level(ens_handle, ens_size, qty, lon_index, lat_index, lev_index, &
521 vals1, my_status)
522 if (my_status /= 0) return
523 call get_values_from_single_level(ens_handle, ens_size, qty, prev_lon, lat_index, lev_index, &
524 vals2, my_status)
525 if (my_status /= 0) return
526
527 vals = (vals1 + vals2) * 0.5_r8
528
529 ! no stagger - cell centers, or W stagger
530 case default
531 call get_values_from_single_level(ens_handle, ens_size, qty, lon_index, lat_index, lev_index, &
532 vals, my_status)
533 if (my_status /= 0) return
534
535end select

586!-----------------------------------------------------------------------
587!> this routine takes care of getting the actual state values. get_state()
588!> communicates with other MPI tasks and can be expensive.
589!>
590!> all ensemble members have the same horizontal location, but different
591!> ensemble members could have different vertical locations and
592!> so be between different vertical layers. this code tries to do the fewest
593!> calls to get_state by only calling it for levels that are actually needed
594!> and setting all members with those same levels in a single pass.
595!>
596
597subroutine get_values_from_varid(ens_handle, ens_size, lon_index, lat_index, lev_index, varid, &
598 vals, my_status)
599type(ensemble_type), intent(in) :: ens_handle
600integer, intent(in) :: ens_size
601integer, intent(in) :: lon_index
602integer, intent(in) :: lat_index
603integer, intent(in) :: lev_index(ens_size)
604integer, intent(in) :: varid
605real(r8), intent(out) :: vals(ens_size)
606integer, intent(out) :: my_status(ens_size)
607
608integer(i8) :: state_indx
609integer :: i, j
610real(r8) :: temp_vals(ens_size)
611logical :: member_done(ens_size)
612
613character(len=*), parameter :: routine = 'get_values_from_varid:'
614
615! as we get the values for each ensemble member, we set the 'done' flag
616! and a good return code.
617my_status(:) = 12
618member_done(:) = .false.
619
620! start with lev_index(1). get the vals into a temp var.
621! run through 2-N. any other member that has the same level
622! set the outgoing values. keep a separate flag for which
623! member(s) have been done. skip to the next undone member
624! and get the state for that level. repeat until all levels done.
625
626do i=1, ens_size
627
628 if (member_done(i)) cycle
629
630 state_indx = get_dart_vector_index(lon_index, lat_index, lev_index(i), domain_id, varid)
631
632 if (state_indx < 0) then
633 write(string1,*) 'Should not happen: could not find dart state index from '
634 write(string2,*) 'lon, lat, and lev index :', lon_index, lat_index, lev_index
635 call error_handler(e_err,routine,string1,source,revision,revdate,text2=string2)
636 return
637 endif
638
639 temp_vals(:) = get_state(state_indx, ens_handle) ! all the ensemble members for level (i)
640
641 ! start at i, because my ensemble member is clearly at this level.
642 ! then continue on to see if any other members are also at this level.
643 do j=i, ens_size
644 if (member_done(j)) cycle
645
646 if (lev_index(j) == lev_index(i)) then
647 vals(j) = temp_vals(j)
648 member_done(j) = .true.
649 my_status(j) = 0
650 endif
651
652 enddo
653enddo
654
655end subroutine get_values_from_varid

702!-----------------------------------------------------------------------
703!>
704!> Model interpolate will interpolate any DART state variable
705!> to the given location.
706!>
707!> @param state_handle DART ensemble handle
708!> @param ens_size DART ensemble size
709!> @param location the location of interest
710!> @param obs_qty the DART Quantity of interest
711!> @param interp_vals the estimated value of the DART state at the location
712!> of interest (the interpolated value).
713!> @param istatus interpolation status ... 0 == success, /=0 is a failure
714!>
715!> istatus = 2 asked to interpolate an unknown/unsupported quantity
716!> istatus = 3 cannot locate horizontal quad
717!> istatus = 4 cannot locate enclosing vertical levels
718!> istatus = 5 cannot retrieve state vector values
719!> istatus = 6 cannot get values at quad corners
720!> istatus = 7 unused (error code available)
721!> istatus = 8 cannot interpolate in the quad to get the values
722!> istatus = 9 unused (error code available)
723!> istatus = 10 cannot get vertical levels for an obs on pressure levels
724!> istatus = 11 cannot get vertical levels for an obs on height levels
725!> istatus = 12 cannot get values from obs quantity
726!> istatus = 13 can not interpolate values of this quantity
727!> istatus = 14 obs above user-defined assimilation top pressure
728!> istatus = 15 can not get indices from given state vector index
729!> istatus = 16 cannot do vertical interpolation for bottom layer
730!> istatus = 17 cannot do vertical interpolation for top layer
731!> istatus = 98 unknown error - shouldn't happen
732!> istatus = 99 unknown error - shouldn't happen
733!>
734
735subroutine model_interpolate(state_handle, ens_size, location, obs_qty, interp_vals, istatus)
736
737type(ensemble_type), intent(in) :: state_handle
738integer, intent(in) :: ens_size
739type(location_type), intent(in) :: location
740integer, intent(in) :: obs_qty
741real(r8), intent(out) :: interp_vals(ens_size) !< array of interpolated values
742integer, intent(out) :: istatus(ens_size)
743
744character(len=*), parameter :: routine = 'model_interpolate:'
745
746integer :: varid, which_vert, status1
747integer :: four_lons(4), four_lats(4)
748integer :: status_array(ens_size)
749real(r8) :: lon_fract, lat_fract
750real(r8) :: lon_lat_vert(3)
751real(r8) :: quad_vals(ens_size,4)
752type(quad_interp_handle) :: interp_handle ! should this be a pointer??
753 ! is it replicating the internal arrays on assignment?
754
755integer :: imem
756
757if ( .not. module_initialized ) call static_init_model
758
759
760! Successful istatus is 0
761interp_vals(:) = missing_r8
762istatus(:) = 99
763
764! do we know how to interpolate this quantity?
765call ok_to_interpolate(obs_qty, varid, status1)
766
767if (status1 /= 0) then
768 if(debug_level > 12) then
769 write(string1,*)'did not find observation quantity ', obs_qty, ' in the state vector'
770 call error_handler(e_msg,routine,string1,source,revision,revdate)
771 endif
772 istatus(:) = status1 ! this quantity not in the state vector
773 return
774endif
775
776! get the grid handle for the right staggered grid
777interp_handle = get_interp_handle(obs_qty)
778
779! unpack the location type into lon, lat, vert, vert_type
780lon_lat_vert = get_location(location)
781which_vert = nint(query_location(location))
782
783! get the indices for the 4 corners of the quad in the horizontal, plus
784! the fraction across the quad for the obs location
785call quad_lon_lat_locate(interp_handle, lon_lat_vert(1), lon_lat_vert(2), &
786 four_lons, four_lats, lon_fract, lat_fract, status1)
787if (status1 /= 0) then
788 istatus(:) = 3 ! cannot locate enclosing horizontal quad
789 return
790endif
791
792! if we are avoiding assimilating obs above a given pressure, test here and return.
793if (discarding_high_obs) then
794 call obs_too_high(lon_lat_vert(3), which_vert, status1)
795 if (status1 /= 0) then
796 istatus(:) = status1
797 return
798 endif
799endif
800
801call get_quad_vals(state_handle, ens_size, varid, obs_qty, four_lons, four_lats, &
802 lon_lat_vert, which_vert, quad_vals, status_array)
803
804!>@todo FIXME : Here we are failing if any ensemble member fails. Instead
805!> we should be using track status...
806if (any(status_array /= 0)) then
807 istatus(:) = maxval(status_array) ! cannot get the state values at the corners
808 return
809endif
810
811do imem=1,ens_size
812 ! do the horizontal interpolation for each ensemble member.
813 ! call one at a time to avoid creating temporary arrays
814 call quad_lon_lat_evaluate(interp_handle, lon_fract, lat_fract, &
815 quad_vals(imem,:), interp_vals(imem), status_array(imem))
816end do
817
818if (using_chemistry) &
819 interp_vals = interp_vals * get_volume_mixing_ratio(obs_qty)
820
821! all interp values should be set by now. set istatus
822istatus(:) = 0
823
824end subroutine model_interpolate

1621!> loop through all levels to get the mean mass.

1624subroutine compute_mean_mass(ens_handle, ens_size, lon_index, lat_index, nlevels, qty, mbar, istatus)
1625type(ensemble_type), intent(in) :: ens_handle
1626integer, intent(in) :: ens_size
1627integer, intent(in) :: lon_index
1628integer, intent(in) :: lat_index
1629integer, intent(in) :: nlevels
1630integer, intent(in) :: qty
1631real(r8), intent(out) :: mbar(nlevels, ens_size)
1632integer, intent(out) :: istatus
1633
1634integer :: k, this_qty
1635real(r8) :: mmr_o1(ens_size, nlevels), &
1636 mmr_o2(ens_size, nlevels), &
1637 mmr_h1(ens_size, nlevels), &
1638 mmr_n2(ens_size, nlevels)
1639real(r8) :: O_molar_mass, O2_molar_mass, H_molar_mass, N2_molar_mass

1641! do this outside the subroutine? it never changes throughout the
1642! run of the program
1643o_molar_mass = get_molar_mass(qty_atomic_oxygen_mixing_ratio)
1644o2_molar_mass = get_molar_mass(qty_molec_oxygen_mixing_ratio)
1645h_molar_mass = get_molar_mass(qty_atomic_h_mixing_ratio)
1646n2_molar_mass = get_molar_mass(qty_nitrogen)
1647
1648
1649
1650! High topped models (WACCM-X) need to account for the changing composition
1651! of the atmosphere with height. This requires several variables from the
1652! initial file, which may not be available from low topped models.
1653do k = 1, nlevels
1654
1655 this_qty = qty_atomic_oxygen_mixing_ratio
1656 call get_staggered_values_from_qty(ens_handle, ens_size, this_qty, &
1657 lon_index, lat_index, k, qty, mmr_o1(:, k), istatus)
1658 if (istatus /= 0) return

1675 + mmr_o2(:,k)/o2_molar_mass &
1676 + mmr_h1(:,k)/h_molar_mass &
